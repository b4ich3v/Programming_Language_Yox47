function fib(n: int) int {
    if (n == 0)  return 0;
    if (n == 1)  return 1;
    return fib(n - 1) + fib(n - 2);
}

function banner() {
    char c = '=';
    for (int i = 0; i < 10; i = i + 1) {
        print(c);
    }
}

function main() int {
    banner();                     
    int k = 7;
    int fk = fib(k);               
    
    box data[4] = { fk, k, 'X', true };
    print(data);

    choose (fk) {
        case 8:  fk = fk + 80;
        case 13: fk = fk + 130;    
        default: fk = fk + 1;
    }

    int counter = 0;
    while (counter < 3) {
        print(counter);
        counter = counter + 1;
    }

    {
        int fk = 42;
        print(fk);                 
    }

    print(fk);                     
    banner();                      

    return fk;                     
}
